# 深拷贝

```js
    function fn(source) {
        let result;
        if (typeof source === 'object') {
            result = Array.isArray(source) ? [] : {};
            for (let key in source) {
                if (typeof source === 'object') {
                    result[key] = fn(source[key]);
                } else {
                    result[key] = source[key];
                }
            }
        } else {
            result = source;
        }
        return result;
    }
    var obj1 = {
      age: 1,
      name: '达撒',
      say: ['aaa', 'bbb'],
      hi: {
        age: 22,
        name: "打"
      }
    }
    var obj2 = fn(obj1);
```
